        self.nodes["files_to_list"].set_plug_value("file2", traits.Undefined)

        self.add_link("mean_stddev_calc.std_out_files->concat_to_list.list2")
        self.add_link(
            "concat_to_list.out_list->result_collector.parameter_files"
